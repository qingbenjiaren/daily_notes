状态机有 3 个组成部分：状态（State）、事件（Event）、动作（Action）



		public enum State {
		  SMALL(0),
		  SUPER(1),
		  FIRE(2),
		  CAPE(3);

		  private int value;

		  private State(int value) {
			this.value = value;
		  }

		  public int getValue() {
			return this.value;
		  }
		}
